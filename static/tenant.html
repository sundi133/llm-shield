#!/usr/bin/env python3
"""
LangChain + LLM Shield Integration Example
Complete protected agent with guardrails, tool policies, and runtime validation
"""

import os
import json
import requests
from typing import Dict, List, Any, Optional
from langchain.agents import AgentType, initialize_agent, load_tools
from langchain.llms import OpenAI
from langchain.tools import BaseTool
from langchain.schema import AgentAction, AgentFinish
from langchain.callbacks.base import BaseCallbackHandler

class LLMShieldConfig:
    """Configuration for LLM Shield integration"""
    def __init__(self):
        # Support both local and RunPod hosted LLM Shield
        self.base_url = os.getenv('LLM_SHIELD_URL', 'http://localhost:8080')
        self.auth_token = os.getenv('RUNPOD_TOKEN')  # For RunPod hosted
        self.api_key = os.getenv('API_KEY')
        self.tenant_id = os.getenv('TENANT_ID')

        self.session = requests.Session()
        headers = {
            'Content-Type': 'application/json'
        }

        # Configure authentication based on deployment type
        if self.auth_token:
            # RunPod hosted LLM Shield
            headers['Authorization'] = f'Bearer {self.auth_token}'

        if self.api_key:
            # API key for both local and RunPod
            headers['X-API-Key'] = self.api_key

        self.session.headers.update(headers)

class LLMShieldAgentManager:
    """Manages agent registration and tool policies with LLM Shield"""

    def __init__(self, config: LLMShieldConfig):
        self.config = config

    def register_agent(self, agent_config: Dict) -> Dict:
        """Register an agent with LLM Shield"""
        try:
            response = self.config.session.post(
                f"{self.config.base_url}/v1/agents/registry",
                json=agent_config
            )
            if response.status_code in [200, 201]:
                print(f"✅ Agent '{agent_config['id']}' registered successfully")
                return response.json()
            else:
                print(f"❌ Failed to register agent: {response.status_code}")
                return {}
        except Exception as e:
            print(f"❌ Error registering agent: {e}")
            return {}

    def update_tool_policies(self, policies: Dict) -> bool:
        """Update tool policies for all agents"""
        try:
            response = self.config.session.put(
                f"{self.config.base_url}/v1/agents/tools/policies",
                json=policies
            )
            if response.status_code == 200:
                print("✅ Tool policies updated successfully")
                return True
            else:
                print(f"❌ Failed to update tool policies: {response.status_code}")
                return False
        except Exception as e:
            print(f"❌ Error updating tool policies: {e}")
            return False

class GuardrailsCallbackHandler(BaseCallbackHandler):
    """LangChain callback handler for LLM Shield integration"""

    def __init__(self, config: LLMShieldConfig):
        self.config = config

    def on_agent_action(self, action: AgentAction, **kwargs) -> Any:
        """Validate agent actions before execution"""
        print(f"🛡️ Validating action: {action.tool}")

        # Validate tool call with LLM Shield
        validation_payload = {
            "tool_name": action.tool,
            "parameters": {"input": action.tool_input},
            "agent_id": kwargs.get('agent_id', 'langchain_agent'),
            "tenant_id": self.config.tenant_id
        }

        try:
            response = self.config.session.post(
                f"{self.config.base_url}/v1/shield/tool/check",
                json=validation_payload
            )

            if response.status_code == 200:
                result = response.json()
                if not result.get('allowed', True):
                    print(f"🚫 Tool call blocked: {result.get('reason', 'Policy violation')}")
                    return None
                else:
                    print(f"✅ Tool call approved: {action.tool}")
            else:
                print(f"⚠️ Could not validate tool call: {response.status_code}")
        except Exception as e:
            print(f"⚠️ Guardrails validation error: {e}")

class SafeLangChainAgent:
    """LangChain agent with integrated LLM Shield protection"""

    def __init__(self, agent_id: str, agent_config: Dict):
        self.agent_id = agent_id
        self.config = LLMShieldConfig()
        self.agent_manager = LLMShieldAgentManager(self.config)
        self.callback_handler = GuardrailsCallbackHandler(self.config)

        # Register agent with LLM Shield
        self.agent_manager.register_agent(agent_config)

        # Initialize LangChain components
        self.llm = OpenAI(temperature=0)
        self.tools = []
        self.agent = None

    def add_tool(self, tool_name: str, tool_policy: Dict = None):
        """Add a tool with optional policy"""
        # Load LangChain tool
        if tool_name in ['serpapi', 'llm-math', 'requests', 'terminal']:
            langchain_tools = load_tools([tool_name], llm=self.llm)
            self.tools.extend(langchain_tools)

        # Set tool policy if provided
        if tool_policy:
            self.agent_manager.set_tool_policy(tool_name, tool_policy)
            print(f"📋 Policy set for tool: {tool_name}")

    def initialize_agent(self):
        """Initialize the LangChain agent with guardrails"""
        self.agent = initialize_agent(
            tools=self.tools,
            llm=self.llm,
            agent=AgentType.ZERO_SHOT_REACT_DESCRIPTION,
            callbacks=[self.callback_handler],
            verbose=True
        )
        print(f"🤖 Agent '{self.agent_id}' initialized with {len(self.tools)} tools")

    def run(self, query: str) -> str:
        """Run the agent with full guardrails protection"""
        if not self.agent:
            self.initialize_agent()

        # Validate input query
        input_validation = {
            "message": query,
            "tenant_id": self.config.tenant_id
        }

        try:
            response = self.config.session.post(
                f"{self.config.base_url}/guardrails/input",
                json=input_validation
            )

            if response.status_code == 200:
                result = response.json()
                if not result.get('safe', True):
                    triggered = [g['guardrail'] for g in result.get('guardrail_results', []) if not g.get('passed', True)]
                    return f"🚫 Query blocked by input guardrails: {', '.join(triggered)}"

            # Execute agent with protection
            print(f"🚀 Executing agent query: {query}")
            result = self.agent.run(query)
            return result

        except Exception as e:
            print(f"❌ Agent execution error: {e}")
            return f"Error: {str(e)}"

# Example usage
def main():
    """Example usage of LangChain + LLM Shield integration"""

    # 1. Define agent configuration
    financial_agent_config = {

#!/usr/bin/env python3
"""
CrewAI + LLM Shield Integration Example
Multi-agent crew with protected agents and unified policy enforcement
"""

import os
import json
import requests
from typing import Dict, List, Any, Optional
from crewai import Agent, Task, Crew, Process
from crewai_tools import SerperDevTool, WebsiteSearchTool, FileReadTool
from langchain.llms import OpenAI

class LLMShieldCrewManager:
    """Manages CrewAI integration with LLM Shield"""

    def __init__(self):
        # Support both local and RunPod hosted LLM Shield
        self.base_url = os.getenv('LLM_SHIELD_URL', 'http://localhost:8080')
        self.auth_token = os.getenv('RUNPOD_TOKEN')  # For RunPod hosted
        self.api_key = os.getenv('API_KEY')
        self.tenant_id = os.getenv('TENANT_ID')

        self.session = requests.Session()
        headers = {
            'Content-Type': 'application/json'
        }

        # Configure authentication based on deployment type
        if self.auth_token:
            # RunPod hosted LLM Shield
            headers['Authorization'] = f'Bearer {self.auth_token}'

        if self.api_key:
            # API key for both local and RunPod
            headers['X-API-Key'] = self.api_key

        self.session.headers.update(headers)

    def register_crew_agents(self, crew_config: Dict) -> bool:
        """Register all agents in a crew with LLM Shield"""
        success = True
        for agent_config in crew_config.get('agents', []):
            try:
                response = self.session.post(
                    f"{self.base_url}/v1/agents/registry",
                    json=agent_config
                )
                if response.status_code in [200, 201]:
                    print(f"✅ Agent '{agent_config['id']}' registered")
                else:
                    print(f"❌ Failed to register agent '{agent_config['id']}': {response.status_code}")
                    success = False
            except Exception as e:
                print(f"❌ Error registering agent '{agent_config['id']}': {e}")
                success = False
        return success

    def validate_input(self, message: str) -> Dict:
        """Validate input against guardrails"""
        payload = {
            "message": message,
            "tenant_id": self.tenant_id
        }

        try:
            response = self.session.post(f"{self.base_url}/guardrails/input", json=payload)
            if response.status_code == 200:
                return response.json()
            else:
                return {"safe": False, "error": f"HTTP {response.status_code}"}
        except Exception as e:
            return {"safe": False, "error": str(e)}

    def validate_output(self, output: str) -> Dict:
        """Validate output against guardrails"""
        payload = {
            "output": output,
            "tenant_id": self.tenant_id
        }

        try:
            response = self.session.post(f"{self.base_url}/guardrails/output", json=payload)
            if response.status_code == 200:
                return response.json()
            else:
                return {"safe": False, "error": f"HTTP {response.status_code}"}
        except Exception as e:
            return {"safe": False, "error": str(e)}

class ProtectedCrewAI:
    """CrewAI with integrated LLM Shield protection"""

    def __init__(self, crew_name: str):
        self.crew_name = crew_name
        self.shield_manager = LLMShieldCrewManager()
        self.llm = OpenAI(temperature=0.1)
        self.agents = []
        self.tasks = []
        self.crew = None

    def create_protected_agent(self, agent_config: Dict) -> Agent:
        """Create a CrewAI agent with LLM Shield protection"""

        # Register with LLM Shield first
        shield_config = {
            "id": agent_config["id"],
            "name": agent_config["role"],
            "role": agent_config.get("role_type", "assistant"),
            "description": agent_config["goal"],
            "tools": [tool.name if hasattr(tool, 'name') else str(tool)
                     for tool in agent_config.get("tools", [])],
            "policies": agent_config.get("policies", [])
        }

        response = self.shield_manager.session.post(
            f"{self.shield_manager.base_url}/v1/agents/registry",
            json=shield_config
        )

        if response.status_code in [200, 201]:
            print(f"✅ Agent '{agent_config['id']}' registered with LLM Shield")
        else:
            print(f"⚠️ Could not register agent with LLM Shield: {response.status_code}")

        # Create CrewAI agent
        agent = Agent(
            role=agent_config["role"],
            goal=agent_config["goal"],
            backstory=agent_config.get("backstory", ""),
            tools=agent_config.get("tools", []),
            llm=self.llm,
            verbose=True,
            max_iter=agent_config.get("max_iterations", 5),
            memory=agent_config.get("memory", True),
            allow_delegation=agent_config.get("allow_delegation", False)
        )

        # Store agent with metadata
        agent_data = {
            "agent": agent,
            "id": agent_config["id"],
            "policies": agent_config.get("policies", [])
        }
        self.agents.append(agent_data)

        return agent

    def create_protected_task(self, task_config: Dict, agent: Agent) -> Task:
        """Create a task with input/output validation"""

        def validate_task_description(description):
            """Validate task description"""
            validation = self.shield_manager.validate_input(description)
            if not validation.get('safe', True):
                triggered = [g['guardrail'] for g in validation.get('guardrail_results', []) if not g.get('passed', True)]
                raise ValueError(f"Task blocked by input guardrails: {', '.join(triggered)}")
            return description

        task = Task(
            description=validate_task_description(task_config["description"]),
            agent=agent,
            expected_output=task_config.get("expected_output", "Detailed analysis and recommendations"),
            tools=task_config.get("tools", [])
        )

        self.tasks.append(task)
        return task

    def setup_tool_policies(self, tool_policies: Dict):
        """Setup tool policies for the crew"""
        try:
            response = self.shield_manager.session.put(
                f"{self.shield_manager.base_url}/v1/agents/tools/policies",
                json=tool_policies
            )
            if response.status_code == 200:
                print("✅ Tool policies configured")
            else:
                print(f"⚠️ Could not configure tool policies: {response.status_code}")
        except Exception as e:
            print(f"⚠️ Error configuring tool policies: {e}")

    def execute_crew(self, process_type: Process = Process.sequential) -> str:
        """Execute the crew with full protection"""

        # Create crew
        self.crew = Crew(
            agents=[agent_data["agent"] for agent_data in self.agents],
            tasks=self.tasks,
            process=process_type,
            verbose=True,
            memory=True
        )

        print(f"🚀 Executing crew '{self.crew_name}' with {len(self.agents)} agents")

        try:
            # Execute crew
            result = self.crew.kickoff()

            # Validate final output
            validation = self.shield_manager.validate_output(str(result))
            if not validation.get('safe', True):
                triggered = [g['guardrail'] for g in validation.get('guardrail_results', []) if not g.get('passed', True)]
                return f"🚫 Output blocked by guardrails: {', '.join(triggered)}"

            print("✅ Crew execution completed successfully")
            return str(result)

        except Exception as e:
            print(f"❌ Crew execution failed: {e}")
            return f"Error: {str(e)}"

    }

    analysis_agent = crew.create_protected_agent(analysis_agent_config)

    # Create protected tasks
    research_task_config = {
        "description": """Research the latest financial performance and market position of Apple Inc. (AAPL).
        Focus on factual, publicly available information from reputable financial sources.""",
        "expected_output": "Comprehensive research report with key financial metrics and market analysis",
        "tools": [search_tool, website_tool]
    }

    analysis_task_config = {
        "description": """Based on the research findings, provide a detailed investment analysis of Apple Inc.
        Ensure all analysis follows regulatory guidelines and professional standards.""",
        "expected_output": "Professional investment analysis with clear recommendation and supporting rationale"
    }

    research_task = crew.create_protected_task(research_task_config, research_agent)
    analysis_task = crew.create_protected_task(analysis_task_config, analysis_agent)

    # Execute the crew
    result = crew.execute_crew(Process.sequential)

    print("\n📋 Final Result:")
    print("-" * 50)
    print(result)

if __name__ == "__main__":
    main()
