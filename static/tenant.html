"""
LangChain + LLM Shield Integration
Guardrails, RBAC, data policies, and shadow agent discovery
"""
import os, json, requests
from langchain_openai import ChatOpenAI
from langchain.agents import create_openai_tools_agent, AgentExecutor
from langchain.tools import tool
from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder

SHIELD_URL = os.getenv("LLM_SHIELD_URL", "http://localhost:8080")
API_KEY    = os.getenv("API_KEY")
AGENT_ID   = os.getenv("AGENT_ID", "my-langchain-agent")
USER_ROLE  = os.getenv("USER_ROLE", "user")

shield = requests.Session()
shield.headers.update({
    "X-API-Key":   API_KEY,
    "X-Agent-Key": AGENT_ID,
    "X-User-Role": USER_ROLE,
    "Content-Type": "application/json",
})


# ── 1. Register Agent (do this once) ──────────────────────────────
def register_agent():
    """Register the agent so Shield knows about it.
    If you skip this, the agent appears as a SHADOW AGENT."""
    resp = shield.post(f"{SHIELD_URL}/v1/agents/registry", json={
        "agent_id": AGENT_ID,
        "name": "LangChain Support Agent",
        "description": "Customer support agent built with LangChain",
        "tools": ["search_faq", "create_ticket", "check_order_status"],
        "role_permissions": {
            "user":    ["search_faq", "check_order_status"],
            "support": ["search_faq", "create_ticket", "check_order_status"],
            "admin":   ["search_faq", "create_ticket", "check_order_status"],
        },
    })
    print("Registered:", resp.json())


# ── 2. Define LangChain tools ─────────────────────────────────────
@tool
def search_faq(query: str) -> str:
    """Search the FAQ knowledge base for answers."""
    return f"FAQ result for '{query}': Check our help center at /help."

@tool
def create_ticket(subject: str, description: str) -> str:
    """Create a support ticket for the customer."""
    return f"Ticket created: {subject} — ID: TKT-{hash(subject) % 10000}"

@tool
def check_order_status(order_id: str) -> str:
    """Check the current status of a customer order."""
    return f"Order {order_id}: Shipped, arriving in 2 days."

lc_tools = [search_faq, create_ticket, check_order_status]


# ── 3. Shielded LangChain agent ──────────────────────────────────
def run_agent(user_message: str):
    # Step 1: Input guardrails
    guard_in = shield.post(f"{SHIELD_URL}/guardrails/input",
                           json={"message": user_message}).json()
    if guard_in.get("action") == "block":
        blocked = [r for r in guard_in.get("guardrail_results", [])
                   if r.get("action") == "block"]
        return f"Blocked by input guardrails: {blocked[0]['guardrail']}"

    # Step 2: Use Shield's agent chat for RBAC + tool execution
    # This sends tools in OpenAI format, gets back RBAC results
    openai_tools = []
    for t in lc_tools:
        openai_tools.append({"type": "function", "function": {
            "name": t.name,
            "description": t.description,
            "strict": True,
            "parameters": {
                "type": "object",
                "properties": {k: {"type": "string"}
                               for k in (t.args_schema.schema()
                                         .get("properties", {}).keys())},
                "required": list(t.args_schema.schema()
                                 .get("required", [])),
                "additionalProperties": False,
            },
        }})

    result = shield.post(f"{SHIELD_URL}/v1/shield/chat/agent", json={
        "messages": [{"role": "user", "content": user_message}],
        "agent_key": AGENT_ID,
        "user_role": USER_ROLE,
        "llm_api_key": os.getenv("OPENAI_API_KEY"),
        "tools": openai_tools,
    }).json()

    # Step 3: Process tool calls with RBAC + data policy enforcement
    output_parts = []
    for tc in result.get("tool_calls", []):
        name = tc["tool_name"]

        # 3a. RBAC check (already in response)
        if not tc["rbac"]["allowed"]:
            output_parts.append(f"BLOCKED: {name} — {tc['rbac']['message']}")
            continue

        # 3b. PRE-EXECUTION: Data policy check on tool arguments
        tool_check = shield.post(f"{SHIELD_URL}/v1/shield/tool/check", json={
            "tool_name": name,
            "tool_input": tc["arguments"],
            "agent_id": AGENT_ID,
            "user_role": USER_ROLE,
        }).json()
        if not tool_check.get("allowed", True):
            output_parts.append(
                f"BLOCKED by data policy: {name} — {tool_check.get('reason', 'policy violation')}")
            continue

        # 3c. Execute the tool
        tool_fn = {t.name: t for t in lc_tools}.get(name)
        if not tool_fn:
            continue
        raw_output = tool_fn.invoke(tc["arguments"])

        # 3d. POST-EXECUTION: Sanitize tool output against data policies
        sanitized = shield.post(f"{SHIELD_URL}/v1/shield/tool/output", json={
            "tool_name": name,
            "tool_output": str(raw_output),
            "agent_id": AGENT_ID,
            "user_role": USER_ROLE,
        }).json()
        if sanitized.get("action") == "block":
            output_parts.append(f"BLOCKED: {name} output violated data policy")
        else:
            clean = sanitized.get("sanitized_output", str(raw_output))
            output_parts.append(f"[{name}] {clean}")

    # Shadow discovery feedback
    if result.get("unregistered"):
        u = result["unregistered"]
        if u.get("agents"):
            output_parts.append(f"[Shadow agent: {u['agents']}]")
        if u.get("tools"):
            output_parts.append(f"[Shadow tools: {u['tools']}]")

    # Step 4: Output guardrails
    final_text = result.get("text", "") or "\n".join(output_parts)
    if final_text:
        guard_out = shield.post(f"{SHIELD_URL}/guardrails/output",
                                json={"output": final_text}).json()
        if guard_out.get("action") == "block":
            return "Response blocked by output policy."

    return final_text or "\n".join(output_parts)


if __name__ == "__main__":
    # register_agent()  # Uncomment on first run
    print(run_agent("What is your return policy?"))
    print(run_agent("Check status of order ORD-555"))
    print(run_agent("Create a ticket: billing issue"))

            

          

        


        

          
3. Shadow Discovery

          
What happens when a LangChain agent is deployed without registering

          

            

              If AGENT_ID is not registered:

              1. Every API call with that X-Agent-Key is tracked by the middleware

              2. Any tools the LLM calls are flagged as shadow tools

              3. The tenant portal Agents tab shows a "Shadow Discovery" panel

              4. Metadata captured: call count, endpoints hit, roles used, first/last seen

              5. Admin can dismiss or register the agent from the UI
            

          

        


        

          
4. Integration Architecture

          

            

              User Message

                │

                ├─▶ Shield /guardrails/input ── block toxic/adversarial/PII

                │

                ├─▶ Shield /v1/shield/chat/agent ── LLM picks tools + RBAC

                │     ├── blocked → deny + log

                │     └── unregistered → shadow discovery

                │

                ├─▶ Shield /v1/shield/tool/check ── data policy on tool args

                │     └── allowed → LangChain tool.invoke()

                │

                ├─▶ Shield /v1/shield/tool/output ── sanitize tool output (PII/secrets)

                │

                ├─▶ Shield /guardrails/output ── block competitors/PII/bias

                │

                └─▶ Response to User

"""
CrewAI + LLM Shield — Multi-Agent with RBAC, Guardrails, Shadow Discovery
Each crew member is a registered Shield agent with its own RBAC policy.
"""
import os, json, requests
from crewai import Agent, Task, Crew, Process
from crewai.tools import BaseTool
from typing import Type
from pydantic import BaseModel, Field

SHIELD_URL = os.getenv("LLM_SHIELD_URL", "http://localhost:8080")
API_KEY    = os.getenv("API_KEY")

shield = requests.Session()
shield.headers.update({
    "X-API-Key":   API_KEY,
    "Content-Type": "application/json",
})


# ── 1. Register all crew agents (do once) ─────────────────────────
def register_crew():
    agents = [
        {
            "agent_id": "research-agent",
            "name": "Research Analyst",
            "description": "Searches public data sources",
            "tools": ["web_search", "document_search"],
            "role_permissions": {
                "analyst": ["web_search", "document_search"],
                "viewer":  ["web_search"],
            },
        },
        {
            "agent_id": "writer-agent",
            "name": "Report Writer",
            "description": "Writes final reports from research",
            "tools": ["generate_report", "send_email"],
            "role_permissions": {
                "analyst": ["generate_report", "send_email"],
                "viewer":  ["generate_report"],
            },
        },
    ]
    for a in agents:
        r = shield.post(f"{SHIELD_URL}/v1/agents/registry", json=a)
        print(f"Register {a['agent_id']}: {r.status_code}")


# ── 2. Shield-aware tool wrapper ──────────────────────────────────
class ShieldTool(BaseTool):
    """Wraps any tool call through Shield's chat/agent for RBAC + data policies."""
    agent_key: str = ""
    user_role: str = "analyst"

    def _run(self, query: str) -> str:
        result = shield.post(f"{SHIELD_URL}/v1/shield/chat/agent", json={
            "messages": [{"role": "user", "content": query}],
            "agent_key": self.agent_key,
            "user_role": self.user_role,
            "llm_api_key": os.getenv("OPENAI_API_KEY"),
        }).json()

        parts = []
        for tc in result.get("tool_calls", []):
            name = tc["tool_name"]

            # RBAC check
            if not tc["rbac"]["allowed"]:
                parts.append(f"BLOCKED: {name} — {tc['rbac']['message']}")
                continue

            # PRE-EXECUTION: Data policy check on tool arguments
            tool_check = shield.post(f"{SHIELD_URL}/v1/shield/tool/check", json={
                "tool_name": name,
                "tool_input": tc.get("arguments", {"query": query}),
                "agent_id": self.agent_key,
                "user_role": self.user_role,
            }).json()
            if not tool_check.get("allowed", True):
                parts.append(f"BLOCKED by data policy: {name} — {tool_check.get('reason')}")
                continue

            # Execute tool (actual implementation goes here)
            tool_output = f"[{name}] executed OK"

            # POST-EXECUTION: Sanitize tool output
            sanitized = shield.post(f"{SHIELD_URL}/v1/shield/tool/output", json={
                "tool_name": name,
                "tool_output": tool_output,
                "agent_id": self.agent_key,
                "user_role": self.user_role,
            }).json()
            if sanitized.get("action") == "block":
                parts.append(f"BLOCKED: {name} output violated data policy")
            else:
                parts.append(sanitized.get("sanitized_output", tool_output))

        # Shadow discovery
        unreg = result.get("unregistered", {})
        if unreg.get("agents"):
            parts.append(f"Shadow agents detected: {unreg['agents']}")
        if unreg.get("tools"):
            parts.append(f"Shadow tools detected: {unreg['tools']}")

        return result.get("text", "") or "\n".join(parts)


# ── 3. Input/output guardrails wrapper ────────────────────────────
def check_input(text: str) -> str | None:
    r = shield.post(f"{SHIELD_URL}/guardrails/input",
                    json={"message": text}).json()
    if r.get("action") == "block":
        blocked = [g["guardrail"] for g in r.get("guardrail_results", [])
                   if g.get("action") == "block"]
        return f"Input blocked: {', '.join(blocked)}"
    return None

def check_output(text: str) -> str | None:
    r = shield.post(f"{SHIELD_URL}/guardrails/output",
                    json={"output": text}).json()
    if r.get("action") == "block":
        blocked = [g["guardrail"] for g in r.get("guardrail_results", [])
                   if g.get("action") == "block"]
        return f"Output blocked: {', '.join(blocked)}"
    return None


# ── 4. Build the crew ─────────────────────────────────────────────
class WebSearchInput(BaseModel):
    query: str = Field(description="Search query")

class WebSearchTool(ShieldTool):
    name: str = "web_search"
    description: str = "Search the web for information"
    args_schema: Type[BaseModel] = WebSearchInput
    agent_key: str = "research-agent"

class GenerateReportInput(BaseModel):
    query: str = Field(description="Report topic and data")

class GenerateReportTool(ShieldTool):
    name: str = "generate_report"
    description: str = "Generate a formatted report"
    args_schema: Type[BaseModel] = GenerateReportInput
    agent_key: str = "writer-agent"

def run_crew(topic: str, user_role: str = "analyst"):
    # Input guardrails
    block = check_input(topic)
    if block:
        return block

    researcher = Agent(
        role="Research Analyst",
        goal=f"Research: {topic}",
        backstory="Senior analyst with deep domain expertise",
        tools=[WebSearchTool(user_role=user_role)],
        verbose=True,
    )
    writer = Agent(
        role="Report Writer",
        goal="Write a clear, actionable report",
        backstory="Technical writer specializing in research reports",
        tools=[GenerateReportTool(user_role=user_role)],
        verbose=True,
    )

    research_task = Task(
        description=f"Research {topic} thoroughly using web search",
        expected_output="Key findings with sources",
        agent=researcher,
    )
    report_task = Task(
        description="Synthesize research into a report",
        expected_output="Formatted report with recommendations",
        agent=writer,
    )

    crew = Crew(
        agents=[researcher, writer],
        tasks=[research_task, report_task],
        process=Process.sequential,
        verbose=True,
    )
    result = str(crew.kickoff())

    # Output guardrails
    block = check_output(result)
    if block:
        return block

    return result


# ── 5. Check for shadow agents ────────────────────────────────────
def list_shadow_items():
    """Pull all unregistered agents/tools the tenant has seen."""
    r = shield.get(f"{SHIELD_URL}/v1/agents/unregistered").json()
    if r.get("agents"):
        print("Shadow agents:", json.dumps(r["agents"], indent=2))
    if r.get("tools"):
        print("Shadow tools:", json.dumps(r["tools"], indent=2))
    return r


if __name__ == "__main__":
    # register_crew()  # Uncomment on first run
    print(run_crew("Analyze Q1 2026 market trends in AI"))
    list_shadow_items()

client = OpenAI()

    "X-Agent-Key": AGENT_ID,
    "X-User-Role": USER_ROLE,
    "Content-Type": "application/json",
})


# ── 1. Register Agent (do this once) ──────────────────────────────

    If you skip this, the agent appears as a SHADOW AGENT."""

        "tools": ["lookup_order", "cancel_order", "get_refund_status"],
        "role_permissions": {
            "user":    ["lookup_order", "get_refund_status"],

            "viewer":  ["lookup_order"],

# ── 2. Define tools (OpenAI function calling format) ──────────────
tools = [
    {"type": "function", "function": {
        "name": "lookup_order",
        "description": "Look up order details by order ID",
        "strict": True,
        "parameters": {
            "type": "object",
            "properties": {"order_id": {"type": "string"}},
            "required": ["order_id"],
            "additionalProperties": False,
        },

# ── 3. Shield helpers ─────────────────────────────────────────────
def check_input_guardrails(message: str) -> dict:
    """Run input through Shield guardrails BEFORE sending to LLM."""
    resp = shield.post(f"{SHIELD_URL}/guardrails/input",
                       json={"message": message})
    return resp.json()

def check_output_guardrails(output: str) -> dict:
    """Run LLM output through Shield guardrails BEFORE returning to user."""
    resp = shield.post(f"{SHIELD_URL}/guardrails/output",
                       json={"output": output})
    return resp.json()

def check_rbac(tool_name: str) -> dict:
    """Check if the current agent+role can use this tool."""
    # Shield's agent_chat does this automatically,
    # but for custom flows you can call the policy endpoint:
    resp = shield.get(f"{SHIELD_URL}/v1/agents/tools/policies/{tool_name}")
    return resp.json()


# ── 4. Shielded agent loop ────────────────────────────────────────
def run_agent(user_message: str):
    # Step 1: Input guardrails
    guard_in = check_input_guardrails(user_message)
    if guard_in.get("action") == "block":
        blocked = [r for r in guard_in.get("guardrail_results", [])
                   if r.get("action") == "block"]
        print(f"BLOCKED by input guardrails: {blocked[0]['guardrail']}")
        return f"Sorry, your message was blocked: {blocked[0]['message']}"

    # Step 2: Call OpenAI with tools
    messages = [
        {"role": "system", "content": "You are a helpful support agent."},
        {"role": "user", "content": user_message},
    ]
    response = client.chat.completions.create(
        model="gpt-4o-mini", messages=messages,
        tools=tools, tool_choice="auto",
    )
    msg = response.choices[0].message

    # Step 3: If tool calls, enforce RBAC + data policies via Shield
    if msg.tool_calls:
        chat_resp = shield.post(f"{SHIELD_URL}/v1/shield/chat/agent",
            json={
                "messages": messages,
                "agent_key": AGENT_ID,
                "user_role": USER_ROLE,
                "llm_api_key": os.getenv("OPENAI_API_KEY"),
                "tools": tools,
            })
        result = chat_resp.json()

        output_parts = []
        for tool_result in result.get("tool_calls", []):
            name = tool_result["tool_name"]

            # 3a. RBAC check
            if not tool_result["rbac"]["allowed"]:
                print(f"  BLOCKED: {name} — {tool_result['rbac']['message']}")
                output_parts.append(f"BLOCKED: {name}")
                continue

            # 3b. PRE-EXECUTION: Data policy check on tool arguments
            tool_check = shield.post(f"{SHIELD_URL}/v1/shield/tool/check", json={
                "tool_name": name,
                "tool_input": tool_result.get("arguments", {}),
                "agent_id": AGENT_ID,
                "user_role": USER_ROLE,
            }).json()
            if not tool_check.get("allowed", True):
                print(f"  BLOCKED by data policy: {name} — {tool_check.get('reason')}")
                output_parts.append(f"BLOCKED by data policy: {name}")
                continue

            # 3c. Execute the tool (your implementation)
            print(f"  ALLOWED: {name}")
            raw_output = execute_tool(name, tool_result.get("arguments", {}))

            # 3d. POST-EXECUTION: Sanitize tool output
            sanitized = shield.post(f"{SHIELD_URL}/v1/shield/tool/output", json={
                "tool_name": name,
                "tool_output": str(raw_output),
                "agent_id": AGENT_ID,
                "user_role": USER_ROLE,
            }).json()
            if sanitized.get("action") == "block":
                output_parts.append(f"BLOCKED: {name} output violated data policy")
            else:
                output_parts.append(sanitized.get("sanitized_output", str(raw_output)))

        # Check for shadow warnings
        if result.get("unregistered"):
            unreg = result["unregistered"]
            if unreg.get("agents"):
                print(f"  SHADOW AGENT detected: {unreg['agents']}")
            if unreg.get("tools"):
                print(f"  SHADOW TOOLS detected: {unreg['tools']}")

        return result.get("text", "") or "\n".join(output_parts)

    # Step 4: Output guardrails on plain text responses
    if msg.content:
        guard_out = check_output_guardrails(msg.content)
        if guard_out.get("action") == "block":
            return "Sorry, the response was blocked by content policy."
        return msg.content

3. Shadow Discovery

          
What happens when a developer deploys an agent WITHOUT registering it

          

            

              If AGENT_ID is not registered:

              1. Every API call with that X-Agent-Key is tracked by the middleware

              2. Any tools the LLM calls are flagged as shadow tools

              3. The tenant portal Agents tab shows a "Shadow Discovery" panel

              4. Metadata captured: call count, endpoints hit, roles used, first/last seen

              5. Admin can dismiss or register the agent from the UI

def register_agent():
    """Register so Shield knows this agent. Skip = shadow agent."""
    shield.post(f"{SHIELD_URL}/v1/agents/registry", json={
        "agent_id": AGENT_ID,
        "name": "Claude Support Agent",
        "tools": ["search_knowledge_base", "create_ticket", "escalate_to_human"],
        "role_permissions": {
            "user":    ["search_knowledge_base"],
            "support": ["search_knowledge_base", "create_ticket"],
            "admin":   ["search_knowledge_base", "create_ticket", "escalate_to_human"],
        },

# ── 3. Shield-wrapped Claude agent ────────────────────────────────
def run_claude_agent(user_message: str):
    # Pre-flight: input guardrails
    guard_in = shield.post(f"{SHIELD_URL}/guardrails/input",
                           json={"message": user_message}).json()
    if guard_in.get("action") == "block":
        blocked = next(r for r in guard_in["guardrail_results"]
                       if r["action"] == "block")
        return f"Blocked: {blocked['message']}"

    # Call Claude with tools
    response = claude.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=1024,
        system="You are a helpful support agent. Use tools when needed.",
        messages=[{"role": "user", "content": user_message}],
        tools=claude_tools,
    )

    # Process response blocks
    results = []
    for block in response.content:
        if block.type == "text":
            # Output guardrails on text
            guard_out = shield.post(f"{SHIELD_URL}/guardrails/output",
                                    json={"output": block.text}).json()
            if guard_out.get("action") == "block":
                results.append("Response blocked by output policy.")
            else:
                results.append(block.text)

        elif block.type == "tool_use":
            # RBAC check via Shield's chat endpoint
            # Convert Claude tool format to OpenAI format for Shield
            openai_tools = [{"type": "function", "function": {
                "name": t["name"],
                "description": t["description"],
                "strict": True,
                "parameters": {**t["input_schema"], "additionalProperties": False},
            }} for t in claude_tools]

            chat_resp = shield.post(f"{SHIELD_URL}/v1/shield/chat/agent",
                json={
                    "messages": [
                        {"role": "system", "content": "You are a support agent."},
                        {"role": "user", "content": user_message},
                    ],
                    "agent_key": AGENT_ID,
                    "user_role": USER_ROLE,
                    "llm_api_key": os.getenv("OPENAI_API_KEY"),
                    "tools": openai_tools,
                }).json()

            for tc in chat_resp.get("tool_calls", []):
                name = tc["tool_name"]

                # RBAC check
                if not tc["rbac"]["allowed"]:
                    results.append(f"BLOCKED: {name} — {tc['rbac']['message']}")
                    continue

                # PRE-EXECUTION: Data policy check on tool arguments
                tool_check = shield.post(f"{SHIELD_URL}/v1/shield/tool/check", json={
                    "tool_name": name,
                    "tool_input": block.input,  # Claude's tool_use input
                    "agent_id": AGENT_ID,
                    "user_role": USER_ROLE,
                }).json()
                if not tool_check.get("allowed", True):
                    results.append(f"BLOCKED by data policy: {name} — "
                                   f"{tool_check.get('reason')}")
                    continue

                # Execute your actual tool logic here
                raw_output = execute_tool(name, block.input)

                # POST-EXECUTION: Sanitize tool output
                sanitized = shield.post(f"{SHIELD_URL}/v1/shield/tool/output", json={
                    "tool_name": name,
                    "tool_output": str(raw_output),
                    "agent_id": AGENT_ID,
                    "user_role": USER_ROLE,
                }).json()
                if sanitized.get("action") == "block":
                    results.append(f"BLOCKED: {name} output violated data policy")
                else:
                    results.append(sanitized.get("sanitized_output", str(raw_output)))

            # Shadow discovery feedback
            if chat_resp.get("unregistered"):
                u = chat_resp["unregistered"]
                if u.get("agents"):
                    results.append(f"[Shadow agent detected: {u['agents']}]")

    return "\n".join(results)


if __name__ == "__main__":

    print(run_claude_agent("Search for how to reset my password"))
    print(run_claude_agent("Create an urgent ticket: billing error"))
    print(run_claude_agent("Escalate this to a manager immediately"))
